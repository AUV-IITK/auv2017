66 int detect(cv::Mat image)
67 {
68  cv::Size size(640, 480); // the dst image size,e.g.100x100
69  cv::Mat resizeimage; // dst image
70  cv::Mat bgr_image;
71  resize(image, resizeimage, size); // resize image
72  cv::waitKey(20);
73  // detect red color here
74  medianBlur(resizeimage, bgr_image, 3); // blur to reduce noise
75  // Convert input image to HSV
76  cv::Mat hsv_image;
77  cvtColor(bgr_image, hsv_image, cv::COLOR_BGR2HSV);
78  // keep only red color
79  inRange(hsv_image, cv::Scalar(0, 100, 100), cv::Scalar(179, 255, 255), red_hue_image);
80  GaussianBlur(red_hue_image, red_hue_image, cv::Size(9, 9), 2, 2); // gaussian blur to remove false positives
81  int nonzero = countNonZero(red_hue_image);
82  int nonzeropercentage = nonzero / 3072;
83  if (nonzero > (3072 * percentage)) // return 1 if a major portion of image
84  // has red color, Note : here the size of
85  // image is 640X480 = 307200.
86  return 1;
87  return 0;
88 }
